# ── Parsing PlantEye ──────────────────────────────────────────────────────────
parse_planteye <- function(filepath) {
  df <- read.csv(filepath, check.names = FALSE, stringsAsFactors = FALSE,
                 encoding = "UTF-8")
  df$date <- as.Date(substr(df$timestamp, 1, 10))
  df$posX <- as.integer(sapply(strsplit(df$unit, ":"), `[`, 2))
  df$posY <- as.integer(sapply(strsplit(df$unit, ":"), `[`, 3))
  df
}

# ── Classificazione parametri ─────────────────────────────────────────────────
classify_params <- function(df) {
  meta   <- c("unit","timestamp","sensor","genotype","g_alias","treatment",
               "date","posX","posY")
  all_p  <- setdiff(colnames(df), meta)
  is_dist <- grepl("\\[.+\\]", all_p)
  list(main = all_p[!is_dist], dist = all_p[is_dist], all = all_p)
}

# ── Aggregazione giornaliera per pianta ───────────────────────────────────────
daily_mean_per_plant <- function(df, params) {
  id_cols <- c("unit","date","treatment","posX","posY","genotype","sensor")
  id_cols <- id_cols[id_cols %in% colnames(df)]
  keep    <- c(id_cols, params)
  sub     <- df[, keep[keep %in% colnames(df)], drop = FALSE]
  for (p in params) sub[[p]] <- suppressWarnings(as.numeric(sub[[p]]))
  agg <- aggregate(sub[, params, drop=FALSE],
                   by = sub[, id_cols, drop=FALSE],
                   FUN = mean, na.rm = TRUE)
  agg
}

# ── Aggregazione media±SE per trattamento × giorno ───────────────────────────
agg_trt_day <- function(daily_df, params, selected_units) {
  sub <- daily_df[daily_df$unit %in% selected_units, , drop = FALSE]
  if (!nrow(sub)) return(NULL)
  result <- list()
  for (p in params) {
    vals <- suppressWarnings(as.numeric(sub[[p]]))
    tmp  <- sub; tmp$val <- vals
    m    <- aggregate(val ~ date + treatment, data = tmp, FUN = mean, na.rm = TRUE)
    s    <- aggregate(val ~ date + treatment, data = tmp, FUN = function(x) {
      n <- sum(!is.na(x)); if (n < 2) 0 else sd(x, na.rm=TRUE)/sqrt(n)
    })
    names(m)[3] <- "mean"; names(s)[3] <- "se"
    merged <- merge(m, s, by = c("date","treatment"))
    merged$param <- p
    result[[p]] <- merged
  }
  do.call(rbind, result)
}

# ── Generazione SVG mappa bench ──────────────────────────────────────────────
make_bench_svg <- function(map_df, trt_colors, selected_units,
                            cell_w = 60, cell_h = 60,
                            pad_x = 52, pad_y = 44, legend_h = 40,
                            is_dark = FALSE) {
  fg     <- if (is_dark) "#D8E8D5" else "#374151"
  bg     <- if (is_dark) "#1E261D" else "#FFFFFF"
  n_col  <- max(map_df$posX)
  n_row  <- max(map_df$posY)
  svg_w  <- n_col * cell_w + pad_x * 2
  svg_h  <- n_row * cell_h + pad_y + 16 + legend_h + 24
  rx     <- 10

  cells <- character(0)
  # Etichette assi
  for (x in seq_len(n_col)) {
    cx <- pad_x + (x-1)*cell_w + cell_w/2
    cells <- c(cells, sprintf(
      '<text x="%.0f" y="%.0f" text-anchor="middle" font-size="11" fill="%s" font-family="system-ui">%d</text>',
      cx, pad_y - 8, fg, x))
  }
  for (y in seq_len(n_row)) {
    cy <- pad_y + (y-1)*cell_h + cell_h/2 + 4
    cells <- c(cells, sprintf(
      '<text x="%.0f" y="%.0f" text-anchor="end" font-size="11" fill="%s" font-family="system-ui">%d</text>',
      pad_x - 6, cy, fg, y))
  }

  # Vasi
  for (i in seq_len(nrow(map_df))) {
    uid  <- map_df$unit[i]
    trt  <- map_df$treatment[i]
    sel  <- uid %in% selected_units
    col  <- trt_colors[trt]
    fill <- if (sel) col else "#D1D5DB"
    strk <- if (sel) col else "#9CA3AF"
    sdark_fill <- if (sel) col else "#4A5568"
    sdark_strk <- if (sel) col else "#718096"
    x_svg <- pad_x + (map_df$posX[i]-1)*cell_w + 3
    y_svg <- pad_y + (map_df$posY[i]-1)*cell_h + 3
    safe_id <- gsub(":", "_", uid)
    cells <- c(cells, sprintf(
      '<rect class="vaso%s" id="v_%s" x="%.0f" y="%.0f" width="%.0f" height="%.0f"
       rx="%d" fill="%s" stroke="%s" stroke-width="2.5" style="cursor:pointer;transition:all .12s;"
       onclick="toggleVaso(\'%s\')"
       data-unit="%s" data-trt="%s" data-sel="%s">
         <title>%s&#10;%s</title>
       </rect>',
      if (sel) " sel" else " excl",
      safe_id, x_svg, y_svg, cell_w-6, cell_h-6,
      rx, fill, strk,
      safe_id, uid, trt, if (sel) "1" else "0",
      uid, trt))
    # Mini etichetta unit
    cells <- c(cells, sprintf(
      '<text x="%.0f" y="%.0f" text-anchor="middle" font-size="11"
       fill="%s" font-family="system-ui" style="pointer-events:none;font-weight:600;">%s</text>',
      x_svg + (cell_w-6)/2, y_svg + (cell_h-6)/2 + 4,
      if (sel) "#FFFFFF" else "#9CA3AF",
      uid))
  }

  # Legenda trattamenti
  trts    <- names(trt_colors)
  item_w  <- 115
  total_w <- length(trts) * item_w
  lx0     <- max(4, (svg_w - total_w) / 2)
  ly      <- pad_y + n_row * cell_h + 14
  for (i in seq_along(trts)) {
    col <- trt_colors[trts[i]]
    lx  <- lx0 + (i-1)*item_w
    cells <- c(cells, sprintf(
      '<rect x="%.0f" y="%.0f" width="14" height="14" rx="3" fill="%s"/>
       <text x="%.0f" y="%.0f" font-size="13" font-weight="600" fill="%s" font-family="system-ui">%s</text>',
      lx, ly, col, lx+20, ly+11, fg, trts[i]))
  }

  sprintf('<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 %d %d" width="100%%" preserveAspectRatio="xMidYMid meet" style="background:%s;border-radius:8px;display:block;">%s</svg>',
          svg_w, svg_h + 4, bg, paste(cells, collapse=""))
}

# ── Grafico temporale ─────────────────────────────────────────────────────────
draw_time_chart <- function(trt_day_df, params, trt_colors, show_ribbon = TRUE,
                             ylab = "Mean value") {
  n_p   <- length(params)
  n_col <- 2; n_row <- ceiling(n_p / n_col)
  par(bg="#FFFFFF", col.axis="#1A1A1A", col.lab="#1A1A1A", col.main="#1A1A1A",
      fg="#1A1A1A", font.main=2, cex.main=1.35, cex.lab=1.2, cex.axis=1.1)
  if (n_p > 1) {
    par(mfrow=c(n_row, n_col), mar=c(7.5, 6, 3, 1.5), mgp=c(4, 0.9, 0))
  } else {
    par(mfrow=c(1,1), mar=c(7.5, 6, 3, 1.5), mgp=c(4, 0.9, 0))
  }

  trts      <- intersect(names(trt_colors), unique(trt_day_df$treatment))
  trts      <- trts[order(trts)]
  all_dates <- sort(unique(trt_day_df$date))
  date_ticks <- seq(min(all_dates), max(all_dates), by = 7)

  for (p in params) {
    sub <- trt_day_df[trt_day_df$param == p, , drop=FALSE]
    if (!nrow(sub)) next

    ylim <- range(sub$mean + sub$se, sub$mean - sub$se, na.rm=TRUE)
    ylim[1] <- max(0, ylim[1])
    ylim <- ylim + diff(ylim) * c(-0.05, 0.08)

    p_clean <- gsub("\\s*$", "", p)
    last_tok <- regmatches(p_clean, regexpr("[^ ]+$", p_clean))
    unit_m <- if (length(last_tok) && !grepl("^[[:digit:]%\\[]", last_tok)) last_tok else ""

    plot.new()
    plot.window(xlim=range(all_dates), ylim=ylim)
    rect(par("usr")[1], par("usr")[3], par("usr")[2], par("usr")[4],
         col="#FFFFFF", border=NA)
    abline(h=pretty(ylim), col="#E5E5E5", lwd=0.9)
    abline(v=date_ticks, col="#F0F0F0", lwd=0.7)

    for (trt in trts) {
      d <- sub[sub$treatment==trt, ]
      if (!nrow(d)) next
      d <- d[order(d$date),]
      col <- trt_colors[trt]
      if (show_ribbon) {
        rib_col <- adjustcolor(col, alpha.f=0.18)
        polygon(c(d$date, rev(d$date)),
                c(d$mean + d$se, rev(pmax(d$mean - d$se, 0))),
                col=rib_col, border=NA)
      }
      lines(d$date, d$mean, col=col, lwd=2.2)
    }

    axis(2, las=1, cex.axis=1.1, col="#1A1A1A", col.axis="#1A1A1A",
         mgp=c(4, 0.9, 0))
    box(col="#1A1A1A", lwd=1)

    # Etichette asse X ruotate di 30 gradi
    usr <- par("usr")
    y_off <- (usr[4] - usr[3]) * 0.03
    text(x=date_ticks, y=usr[3] - y_off,
         labels=format(date_ticks, "%d %b"),
         srt=30, adj=c(1,1), xpd=TRUE, cex=1.0, col="#1A1A1A")

    # Asse Y: nome parametro + unità
    ylab_full <- if (nchar(unit_m) > 0) paste0(ylab, " [", trimws(unit_m), "]") else ylab
    title(main=p, col.main="#1A1A1A")
    title(ylab=ylab_full, col.lab="#1A1A1A")

    # Legenda solo nel primo pannello, testo e simboli più grandi
    if (p == params[1]) {
      legend("topleft", legend=trts, col=trt_colors[trts], lwd=2.5,
             bty="n", cex=1.05, pt.cex=1.5)
    }
  }
  par(mfrow=c(1,1))
}

# ── Boxplot per data singola ──────────────────────────────────────────────────
draw_boxplot_day <- function(daily_df, params, trt_colors, selected_date,
                              ylab = "Value", show_jitter = FALSE) {
  sub <- daily_df[daily_df$date == selected_date, , drop = FALSE]
  if (!nrow(sub)) return(invisible(NULL))

  n_p   <- length(params)
  n_col <- 2; n_row <- ceiling(n_p / n_col)
  par(bg="#FFFFFF", col.axis="#1A1A1A", col.lab="#1A1A1A", col.main="#1A1A1A",
      fg="#1A1A1A", font.main=2, cex.main=1.3, cex.lab=1.2, cex.axis=1.1)
  if (n_p > 1) {
    par(mfrow=c(n_row, n_col), mar=c(7, 6, 3.5, 1.5), mgp=c(4, 0.9, 0))
  } else {
    par(mfrow=c(1,1), mar=c(7, 6, 3.5, 1.5), mgp=c(4, 0.9, 0))
  }
  trts <- intersect(names(trt_colors), unique(sub$treatment))
  trts <- trts[order(trts)]

  for (p in params) {
    vals <- suppressWarnings(as.numeric(sub[[p]]))
    sub$val <- vals
    bp_data <- lapply(trts, function(trt) {
      v <- sub$val[sub$treatment == trt]
      v[!is.na(v)]
    })
    names(bp_data) <- trts
    non_empty <- sapply(bp_data, length) > 0
    bp_data <- bp_data[non_empty]
    cols    <- trt_colors[names(bp_data)]

    p_clean  <- gsub("\\s*$", "", p)
    last_tok <- regmatches(p_clean, regexpr("[^ ]+$", p_clean))
    unit_m   <- if (length(last_tok) && !grepl("^[[:digit:]%\\[]", last_tok)) last_tok else ""
    ylab_full <- if (nchar(unit_m) > 0) paste0(ylab, " [", trimws(unit_m), "]") else ylab

    all_vals <- unlist(bp_data); all_vals <- all_vals[!is.na(all_vals)]
    ylim <- if (length(all_vals)) range(all_vals) * c(0.95, 1.05) else c(0,1)

    plot.new()
    plot.window(xlim=c(0.5, length(bp_data)+0.5), ylim=ylim)
    rect(par("usr")[1], par("usr")[3], par("usr")[2], par("usr")[4],
         col="#FFFFFF", border=NA)
    y_major <- pretty(ylim)
    y_major <- y_major[y_major >= ylim[1] & y_major <= ylim[2]]
    abline(h=y_major, col="#E5E5E5", lwd=0.9)

    bp_pos <- seq_along(bp_data)
    boxplot(bp_data, add=TRUE, at=bp_pos, col=cols,
            border="#1A1A1A", lwd=1, axes=FALSE,
            outline=TRUE, pch=19, cex=0.6,
            whisklty=1, staplelwd=1.2, boxwex=0.55)

    if (show_jitter) {
      for (j in seq_along(bp_data)) {
        vals_j <- bp_data[[j]]
        if (!length(vals_j)) next
        jx <- bp_pos[j] + runif(length(vals_j), -0.18, 0.18)
        points(jx, vals_j, pch=21,
               bg=adjustcolor("#1A1A1A", alpha.f=0.35),
               col=adjustcolor("#1A1A1A", alpha.f=0.55),
               cex=0.85, lwd=0.6)
      }
    }

    usr <- par("usr")
    y_off <- (usr[4] - usr[3]) * 0.04
    text(x=bp_pos, y=usr[3] - y_off, labels=names(bp_data),
         srt=30, adj=c(1,1), xpd=TRUE, cex=0.95, col="#1A1A1A")

    axis(2, at=y_major, col="#1A1A1A", col.axis="#1A1A1A", cex.axis=1.1,
         las=1, mgp=c(4,0.9,0))
    box(col="#1A1A1A", lwd=1)
    title(main=paste0(p, "  [", format(selected_date, "%d %b %Y"), "]"),
          col.main="#1A1A1A")
    title(ylab=ylab_full, col.lab="#1A1A1A")
  }
  par(mfrow=c(1,1))
}

# ── Pivot long ────────────────────────────────────────────────────────────────
pivot_long_pe <- function(df, params) {
  id_cols    <- c("unit","timestamp","date","treatment","sensor","genotype","posX","posY")
  id_present <- id_cols[id_cols %in% colnames(df)]
  id_data    <- df[, id_present, drop=FALSE]
  long_rows  <- do.call(rbind, lapply(params, function(p) {
    tmp <- id_data; tmp$parameter <- p; tmp$value <- suppressWarnings(as.numeric(df[[p]])); tmp
  }))
  long_rows[order(rep(seq_len(nrow(df)), length(params))),]
}

# ── Base64 encoder ────────────────────────────────────────────────────────────
base64_encode_raw <- function(raw_bytes) {
  alphabet <- strsplit("ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/","")[[1]]
  n <- length(raw_bytes); pad <- (3-n%%3)%%3
  if (pad>0) raw_bytes <- c(raw_bytes, raw(pad))
  m   <- matrix(as.integer(raw_bytes), nrow=3)
  idx <- rbind(bitwShiftR(m[1,],2),
               bitwOr(bitwShiftL(bitwAnd(m[1,],3L),4), bitwShiftR(m[2,],4)),
               bitwOr(bitwShiftL(bitwAnd(m[2,],15L),2), bitwShiftR(m[3,],6)),
               bitwAnd(m[3,],63L))
  out <- matrix(alphabet[idx+1], nrow=4)
  if (pad>=1) out[4,ncol(out)] <- "="
  if (pad==2) out[3,ncol(out)] <- "="
  paste(out, collapse="")
}
